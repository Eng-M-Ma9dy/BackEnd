( ES6 - Ex / Import )

( File1 - Export ) 



    // For Work by ES6 I need to identification the js File as Module

    // I have tow Way
    // ----------------
    // 1. set file name extension index.mjs (Not Use)
    // 2. Go to package.json > Set "type": "module", Under "main": "index.js",
    
    // For Export The Function or Var For Use in Another JS Sheet
    // Set export Before Function Or Var

    export let hours = 8;

    export function calcHours(rate){

        console.log("hours * rate =", hours * rate);
    }

    // -----------------------------------------------------------------------------

    // i Can Do Only One export Default For Import it in Main Sheet By Any Name 

    export default function SayHi(){
        console.log("Export Default : (Hi)")
    }

    // -----------------------------------------------------------------------------

    let firstName = "Mohamed" 
    let secondName = "Magdy"
    let yourAge = 27

    export function myName(){

        console.log("My Name = ", firstName +" "+ secondName)
    }

    export function myAge(){

        console.log("My Age = ", yourAge)
    }

    // -----------------------------------------------------------------------------

    let x = 10
    let y = 20
    let z = 30

    export{
        x,y,z
    }
