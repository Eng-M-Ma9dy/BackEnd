    // I Use Joi For Make Data INput Validation Before set it in Database
    
    // For install Joi >> npm i joi

    export const validation = (schema) => {

        return (req,res,next) => {
    
            // by default validate out after the first error

            // let {error} = signUpSchema.validate(req.body);
    
            // For Show All Error use abortEarly error
    
            let {error} = schema.validate(req.body,{abortEarly:false});
    
            if (error) {
    
                res.json({message:"Validation Error",error})
    
            }else{
    
                next()
            }
    
        }
    
    }

    // I Use jsonwebtoken For Create Token For Every User  
    
    // For install Jwt >> npm i jsonwebtoken

    import jwt from 'jsonwebtoken'

    export const auth = (req,res,next)=>{
    
        let authorization = req.headers["authorization"];
    
        if (!authorization || (authorization && authorization.startsWith("Bearer ") == false )) {
    
            res.json({message:"Invalid Token, You Aren't Authorized"});
    
        }else{
            console.log("Valid Token, Welcome");
    
            let token_Auth = authorization.split(" ")[1];
    
            jwt.verify(token_Auth,'KeyMega',(err,decoded)=>{

                // 'KeyMega': Hash Key
    
                if (err) {
        
                    res.json({message:"Invalid Token, You Aren't Authorized",err});
        
                } else {
        
                    req.userID = decoded.userID
        
                    next()
                }
            })
        }
    
    
    }

    import userModel from '../../../../Database/models/user.model.js'

    // For Replace Password by Hash Password in Database

    // i will use bcrypt
    
    // For install bcrypt >> npm i bcrypt

    import bcrypt from "bcrypt";

    export const signUp = async (req,res,next)=>{

        let {name,email,password,role} = req.body;
    
        let foundedUser = await userModel.findOne({email})

        if(foundedUser){
        
            res.json ({message:"User Already Register"})
        
        }else{

            // For Use bcrypt
            // bcrypt.hash(pass , Number Of Round) : return promise I need To Use Async / await
            // bcrypt.hashSync(pass , Number Of Round) : don't need to use Async / await
    
            let hashedPassword = bcrypt.hashSync( password , 4 )
    
            let addUser = await userModel.insertMany({name,email,password:hashedPassword,role})
            
            res.json({message:"Done",addUser})
    
        }
    
    }

    // I Use jsonwebtoken For Create Token For Every User  
    
    // For install Joi >> npm i jsonwebtoken

    import jwt from 'jsonwebtoken'

    export const signIn = async (req,res)=>{

        let {email,password} = req.body;

        let foundedUser = await userModel.findOne({email})

        if(foundedUser){

            // bcrypt.compare(pass , Number Of Round) : return promise I need To Use Async / await

            // bcrypt.compareSync(pass , Number Of Round) : don't need to use Async / await

            // let passMatched = await bcrypt.compare(password , foundedUser.password)

            let passMatched = bcrypt.compareSync(password , foundedUser.password)

            if(passMatched){

                let token = jwt.sign({userID:foundedUser._id, name:foundedUser.name},'KeyMega',{expiresIn: 60 * 240 }); // 1 Min
                
                // 'KeyMega': Hash Key

                // console.log(token);

                res.json ({message:"LogIn Success",token})

            }else{

                res.json ({message:"Password Incorrect"})
            }

        }else{

            res.json({message:"Email Not Found, You Have To Register"})
        }

    }

    import userModel from '../../../../DB/models/user.model.js'

    import { sendEmail } from '../../../Services/email.js';
    
    import bcrypt from "bcrypt";

    //-----------------------------------------------------------------------------
    
    export const signUp = async (req,res,next)=>{
    
        let {name,age,email,password,CPassword} = req.body;
        
        if (password != CPassword) return res.json ({message:"Password Not Equal Confirmed Password"})
    
        let foundedUser = await userModel.findOne({email})
    
        if(foundedUser) return next(new AppError("User Already Register",400))
    
        let hashedPassword = bcrypt.hashSync(password,parseInt(process.env.HASH_ROUNDED))
    
        let addUser = await userModel.insertMany({name,age,email,password:hashedPassword})
        
        sendEmail({email});
    
        res.status(201).json({message:"User Added",addUser})
    
    }
    
    //-----------------------------------------------------------------------------
    //-----------------------------------------------------------------------------

    export const verify = async (req,res,next)=>{

        let {token} = req.params;
        
        jwt.verify(token,process.env.Email_KEY_TOKEN, async (err,decoded)=>{

            if (err) return res.json({message:"Invalid Token"})

            let VEUpdated = await userModel.findOneAndUpdate({email:decoded.email},{confirmedEmail:true},{new:true});
            
            VEUpdated.password = undefined
            
            res.json({message:"The Mail Verified",VEUpdated})
        })
    }

    //-----------------------------------------------------------------------------
    //-----------------------------------------------------------------------------
    
    import jwt from 'jsonwebtoken'
    
    export const signIn = async (req,res,next)=>{
            
        let {email,password} = req.body;
    
        let foundedUser = await userModel.findOne({email})
        
        if(!foundedUser) return res.json({message:"Email Not Found, You Have To Register First"})
        
        let passMatched = bcrypt.compareSync(password , foundedUser.password)
    
        if(!passMatched) return res.json ({message:"Incorrect Password"})
    
        let token = jwt.sign({userID:foundedUser._id, name:foundedUser.name},process.env.KEY_TOKEN,{expiresIn: '24h'});
    
        res.status(200).json ({message:"LogIn Success",token})
    
    }

    import { nanoid } from 'nanoid'
    

    // in case i use Local storage
    //-------------------------------

    export const fileUpload = () => {

        const storage = multer.diskStorage({
    
            // For Upload File To Destination Folder 
            //--------------------------------------------

            destination: function (req, file, cb) {
    
              cb(null, 'uploads')
    
            },
    
            // For Rename File i received it
            //---------------------------------

            filename: function (req, file, cb) {
    
              // // - For Create New Name For Every File from multer
    
              // // - Created New Name Without extension
    
              // const uniqueSuffix = Date.now() + '-' + Math.round(Math.random() * 1E9)
    
              // cb(null, uniqueSuffix + '-' + file.originalname)
    
              // // Or --------------------------------------------------------------
    
              // // - For Save File By originalname and use nano id for add random name
    
              cb(null, nanoid() + "_" + file.originalname)
    
            }
    
        });
    
        // For Select Type Of File i can received
        //------------------------------------------

        function fileFilter (req, file, cb) {
      
          const acceptedFilesType = ["image/png","image/jpeg","video/mp4"]
      
          let accepted = acceptedFilesType.find((ele) => file.mimetype.startsWith(ele))
            
          if (accepted) return cb(null, true)
      
          cb(null, false)
        
        }
    
      const upload = multer({ storage: storage , fileFilter })
    
      return upload.single("photoImg")
    
      // photoImg : Key Of Img i Received
    
    }
